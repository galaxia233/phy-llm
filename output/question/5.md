\[ U(r) = -\frac{E}{2} - \frac{1}{2}mv^2 = -\frac{1}{2}mv^2 \]
\[ F(r) = \frac{dU}{dr} = -\frac{1}{2}mv^2 \]

\[ r \to \infty, v \to 0 \Rightarrow E = 0 \] （取 \( U(\infty) = 0 \)）

\[ U_{\text{cond}}(T) = \frac{1}{2} RT - \frac{1}{2} RT = \frac{1}{2} RT \]

\[ U_{\text{cond}}(T) = \frac{1}{2} RT \]

\[ q = \frac{1}{2} RT - U_{\text{cond}}(T) \]

\[ = U_{\text{cond}}(T) - U_{\text{cond}}(T) + RT \]

\[ U_{\text{cond}}(T) = U_{\text{cond}}(T) + RT - L(T) \]

\[ q = \frac{1}{2} RT - 4RT + 4RT \]

\[ = \frac{1}{2} RT - 4RT + 4RT \]

\[ q = \frac{1}{2} RT - U_{\text{cond}}(T) \]

\[ = \frac{1}{2} RT - U_{\text{cond}}(T) \]
